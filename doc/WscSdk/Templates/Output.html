def self.full_hd(modifiers={})
  self.merge({
    stream_format:        "audiovideo",
    passthrough_video:    false,
    passthrough_audio:    false,
    aspect_ratio_height:  1080,
    aspect_ratio_width:   1920,
    bitrate_audio:        128,
    bitrate_video:        4000,
    h264_profile:         "high",
    framerate_reduction:  "0",
    keyframes:            "follow_source"
  }, modifiers)
end
