def self.akamai_cupertino(name, secure_ingest, cors, modifiers={})
  self.merge({
    name:                         name,
    provider:                     WscSdk::Enums::WowzaProvider::AKAMAI_CUPERTINO,
    use_secure_ingest:            secure_ingest,
    use_cors:                     cors
  }, modifiers)
end
