def initialize
  @api_key    = ENV["WSC_API_KEY"]
  @access_key = ENV["WSC_API_ACCESS_KEY"]
  @logger     = ::Logger.new(STDOUT)
  @hostname   = WscSdk::HOSTNAME
  @version    = WscSdk::PATH_VERSION
end
